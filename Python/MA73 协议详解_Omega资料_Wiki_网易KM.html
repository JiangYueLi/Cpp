    MA73协议结构

    USTRUCT(BlueprintType)
struct FCL_LOGIN_GUEST_AUTH_SYN : public FBaseProtocol
{
	UPROPERTY(EditAnywhere, BlueprintReadWrite)	EOSCode m_OSCode;
	UPROPERTY(EditAnywhere, BlueprintReadWrite)	FString m_GuestID;
	UPROPERTY(EditAnywhere, BlueprintReadWrite)	FLoginUserInfo m_LoginUserInfo;

	PROTOCOL_CONSTRUCTOR(FCL_LOGIN_GUEST_AUTH_SYN);
	GENERATED_USTRUCT_BODY() PROTOCOL_VARS(m_OSCode, m_GuestID, m_LoginUserInfo);
};

    以FCL_LOGIN_GUEST_AUTH_SYN 为例子，可以看出一条协议在73中是如何被定义的。

    1. FCL_LOGIN_GUEST_AUTH_SYN 继承FBaseProtocol
    2. 定义协议数据结构，这边定义了3个，分别是m_OSCode，m_GuestID，m_LoginUserInfo
    3. 使用宏定义PROTOCOL_CONSTRUCTOR(FCL_LOGIN_GUEST_AUTH_SYN);
    4. 使用宏定义GENERATED_USTRUCT_BODY()
    5. 使用宏定义PROTOCOL_VARS(m_OSCode, m_GuestID, m_LoginUserInfo);~~~~

    接下来就一步步说明这5步分别是干嘛的

    继承FBaseProtocol

    USTRUCT(BlueprintType)
struct FBaseProtocol
{
	int32 m_Size;    //协议长度，预分配空间，后续会计算真实长度并填充
	uint64 m_HashCode;  // 协议HashCode
	int32 m_TagNumber;  // 协议在双方交互的ID，有点像TCP中的ACK的意思。

	FBaseProtocol() : m_Size(0), m_HashCode(0), m_TagNumber(0) {}
	virtual ~FBaseProtocol() {}

	virtual void Serialize(CSerializer& rSerializer)  //序列化函数
	{
		rSerializer.TSerialize(*this);
	}
	virtual void DeSerialize(CDeSerializer& rDeSerializer)  //反序列化函数
	{
		rDeSerializer.TDeserialize(*this);
	}
	virtual FString GetName()  // 获取协议名称
	{
		return TEXT("FBaseProtocol");
	}
    ......

	GENERATED_USTRUCT_BODY()
		BIND_MEM_FUNC_GET_TIE(m_Size, m_HashCode, m_TagNumber);
};

    FBaseProtocol 结构也很简单，只有3个成员变量，然后定义了序列化和反序列化函数。同时也使用了宏定义。

    定义协议内容

    主要就是看协议要发送什么内容，默认情况下协议支持结构：

    1. int（实际上支持浮点，但不建议）
    2. string、wstring
    3. vector
    4. map
      如果是自定义类型，那么就需要使用到上面的宏定义了。

    PROTOCOL_CONSTRUCTOR()宏定义

    # define PROTOCOL_CONSTRUCTOR(PacketName)\
	PacketName()\
	{\
		m_HashCode = MHash::Str(#PacketName);\
	}\
	virtual ~PacketName()\
	{\
	}\
 	virtual FString GetName()\
 	{\
 		return TEXT(#PacketName); \
 	}\
	virtual FString GetJsonString()\
	{\
		return ObjectConverter::StructToJsonString(*this);\
	}\

    PROTOCOL_CONSTRUCTOR 宏定义内容比较简单，目的是为了提供构造函数，在对象被构造的时候通过协议名称算出HashCode。

    GENERATED_USTRUCT_BODY

    # define GENERATED_USTRUCT_BODY(...)

    这部分在服务器上的宏定义是空的，和UE4有关。

    PROTOCOL_VARS（） 和 PROTOCOL_NOVARS（）

    这个宏定义分为两类，有参数的PROTOCOL_VARS 和 无参数的PROTOCOL_NOVARS

    # define PROTOCOL_NOVARS()\
public:\
	virtual void Serialize(CSerializer& rSerializer)\
	{\
		FBaseProtocol::Serialize(rSerializer); \    //无参数的版本只调用父类的序列化函数，
		\
		m_Size = (int32)rSerializer.GetCurSize();\      //获取协议长度
		rSerializer.OverWrite(0, &m_Size, sizeof(m_Size));\  // 重新将父对象预留的长度填充
		\
		UVATToolPost(true, true);\   
	}\
	virtual void DeSerialize(CDeSerializer& rDeSerializer)\
	{\
		FBaseProtocol::DeSerialize(rDeSerializer); \   // 反序列化也直接对父对象进行
		\
		UVATToolPost(false, true);\
	}

# define PROTOCOL_VARS(...)\
public:\
	virtual void Serialize(CSerializer& rSerializer)\
	{\
		FBaseProtocol::Serialize(rSerializer); \     //有参数版本是先序列化父对象
		rSerializer.TSerialize(*this);\     //然后序列化子对象
		\
		m_Size = (int32)rSerializer.GetCurSize();\  // 获取长度
		rSerializer.OverWrite(0, &m_Size, sizeof(m_Size));\  // 重新填充父对象预留的长度
		\
		UVATToolPost(true, false);\
	}\
	virtual void DeSerialize(CDeSerializer& rDeSerializer)\
	{\
		FBaseProtocol::DeSerialize(rDeSerializer); \   // 反序列化也是先解父对象
		rDeSerializer.TDeserialize(*this);\  //再反序列化子对象
		\
		UVATToolPost(false, false);\
	}\
	BIND_MEM_FUNC_GET_TIE(__VA_ARGS__);  // 关键，定义解压包顺序

    看宏定义代码，实际上有参数和无参数的版本做的事情都差不多，无参数的版本直接对父对象进行操作即可，有参数的版本主要使用 BIND_MEM_FUNC_GET_TIE这个宏定义定义了序列化的顺序。

    至此，如果只是简单的使用73协议，那么使用以上5个部分基本上就对协议有所了解了，可以按照模板进行使用了。接下来会比较详细介绍BIND_MEM_FUNC_GET_TIE 这个宏定义 和 Serialize 的细节了。

    BIND_MEM_FUNC_GET_TIE

    namespace std
{
	template<class... Args>
	auto c_tie( const Args&... args ) ->decltype( std::tuple<const Args&...>( args... ) )
	{
		return std::tuple<const Args&...>( args... );
	}
}

# define BIND_MEM_FUNC_GET_TIE(...)\
	auto get_tie() const ->decltype( std::c_tie(__VA_ARGS__) )\
	{\
		return std::c_tie( __VA_ARGS__ );\
	}\
	auto get_tie() ->decltype( std::tie( __VA_ARGS__ ) )\
	{\
		return std::tie( __VA_ARGS__  );\
	}\

    BIND_MEM_FUNC_GET_TIE 主要是定义了两个函数，当需要const版本的时候，返回的就是std::c_tie(args)， 其中使用std::tuple来包含const的成员变量；当需要的是非const版本的时候，则使用std::tie， std::tie在C++文档中的返回介绍是：A std::tuple object containing lvalue references. 所以返回的tuple是定义的成员变量的左值引用。

    Serialize(CSerializer& rSerializer)

    Serialize(CSerializer& rSerializer)这个函数实际上是序列化最后会调用的函数，之前通过PROTOCOL_VARS宏定义已经为每个协议类型添加了这个函数，而在这个函数内可以调用的是

    rSerializer.TSerialize(*this);

    接下来就看看rSerializer是个什么东西

    class CSerializer
{
private:
	char* m_pBuffer; // 缓冲区
	uint64 m_curSize;  //当前长度
	uint64 m_maxSize;  // 最大长度

	bool m_IsSuccess;   // 是否序列化成功
    ......
  }

    首先CSerializer这个类是一个包含缓冲区的对象，类成员中没有很特殊的对象。而TSerialize是一个函数模板

       	bool Write(const void* pData, uint64 size)
	{
		if (m_curSize + size > m_maxSize)
		{
			m_IsSuccess = false;
			return false;
		}

		memcpy(m_pBuffer + m_curSize, pData, size);
		m_curSize += size;

		return true;
	}

	template <class T>
	typename std::enable_if<std::is_arithmetic<T>::value>::type TSerialize(const T& rData)
	{
		Write(&rData, sizeof(T));
	}

	template <class T>
	typename std::enable_if<std::is_enum<T>::value>::type TSerialize(const T& rData)
	{
		Write(&rData, 1);
	}

	template <class T>
	typename std::enable_if<std::is_string_type<T>::value>::type TSerialize(const T& rData)
	{
		TSerialize(static_cast<int16>(rData.size()));
		if (rData.size() == 0)
			return;

		Write(rData.c_str(), rData.size() * sizeof(T::value_type));
	}

	template <class T>
	typename std::enable_if<std::is_vector_container<T>::value>::type TSerialize(const T& rContainer)
	{
		TSerialize(static_cast<int16>(rContainer.size()));

		for (const auto& value : rContainer)
		{
			TSerialize(value);
		}
	}

	template <class T>
	typename std::enable_if<std::is_map_container<T>::value>::type TSerialize(const T& rContainer)
	{
		TSerialize(static_cast<int16>(rContainer.size()));

		for (auto it = rContainer.begin(); it != rContainer.end(); it++)
		{
			TSerialize(it->first);
			TSerialize(it->second);
		}
	}

    可以看到，当调用TSerialize的时候，就会根据其调用的参数类型来决定使用哪个函数，这边的std::enable_if<std::is_arithmetic::value>::type 这种形式就是来决定调用函数的；

    1. 简单来说当std::is_arithmetic中T是算术类型时，std::is_arithmetic::value 为true，否则为false。
    2. 当std::enable_if 为true时，std::enable_if::type有定义，为void，否则无定义
    3. 根据C++ 重载决议overload resolution 与 SFINAE 来决定对应类型的调用版本

    所以这边上述TSerialize代码就展现了73中支持的基本数据类型。但这边也很明显看出对基本类型的序列化就是把内存直接序列化后直接发送给对方，当平台不一致的时候是否会发生些什么就不好说了，这边尤其点名std::wstring

    而对于自定义类型，就要看下面的代码

        class SerializeCaller
        {
        private:
            CSerializer& m_rSerializer;

        public:
            SerializeCaller(CSerializer& rSerializer) : m_rSerializer(rSerializer) {}

            template <class T>
            void operator()(const T& rData) const
            {
                m_rSerializer.TSerialize(rData);    // 进到这边的时候，是自定义类型中的一个对象。所以只要按一个对象的序列化方式重新调用TSerialize函数
            }
            };

    template <uint64 N>
    struct ForEach_Tuple
    {
        template <class FuncType, class TupleType>
        static void Func( FuncType& rFunc, TupleType& rTuple )
        {
            rFunc( std::get<std::tuple_size<TupleType>::value - N>( rTuple ) );

            ForEach_Tuple<N - 1>::Func( rFunc, rTuple );
        }
    };

    template <>
    struct ForEach_Tuple < 0 >
    {
        template <class FuncType, class TupleType>
        static void Func( FuncType& rFunc, TupleType& rTuple )
        {
        }
    };

    template <class TupleType, class FuncType>
    void ForEach(TupleType& rTuple, FuncType& rFunc)
    {
        ForEach_Tuple<std::tuple_size<TupleType>::value>::Func(rFunc, rTuple);
    }


	template <class T>
	typename std::enable_if<std::has_mem_func_get_tie<T>::value>::type TSerialize(const T& rData)
	{
		SerializeCaller caller(*this);   //新生成一个SerializeCaller
		auto tie = rData.get_tie();   // 通过get_tie来获取自定义类型的序列化顺序

		ForEach(tie, caller);   // forEach调用SerializeCaller的operator()
	}

    自定义类型的TSerialize有点复杂，

    1. 首先是生成SerializeCaller对象，这个对象只是CSerializer的包装，主要是为了在ForEach中使用。
    2. tie则是由之前定义的BIND_MEM_FUNC_GET_TIE()来返回一个由顺序的tuple。
    3. Foreach主要是为了按顺序一个个获取tie中的对象，并调用SerializeCaller的operator()，而operator()中又调用CSerializer的TSerialize，这边就变成了递归调用。

    DeSerialize(CDeSerializer& rDeSerializer)

    基本过程就和Serialize相反，有兴趣的自己可以再分析一遍

    总结

    总体来说，虽然TSerialize中对于基础对象的序列化方式有点蠢，不同平台运行这段代码可能会出现不同的结果，尤其是wstring在windows平台和linux平台下实现大相径庭。但CSerializer 中对于模板和宏定义的使用还是挺厉害的，感觉还是指得学习的。
